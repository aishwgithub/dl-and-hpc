#include <iostream>
#include<vector>
#include<queue>
#include<omp.h>

using namespace std;

int vertices;
vector<vector<int>> adjList;

void addEdge(int u,int v){
	adjList[u].push_back(v);
	adjList[v].push_back(u);
}

void parallelBFS(int start){
	vector<bool> visited(vertices,false);

	queue<int> q;
	visited[start] = true;
	q.push(start);

	cout<<"BFS: ";

	while(!q.empty()){

		int levelsize = q.size();
		vector<int> currentLevel;

		for(int i = 0;i<levelsize;i++){
			currentLevel.push_back(q.front());
			q.pop();
		}

		//process current level nodes in parallel
		//each thread handles one node from this level

		#pragma omp parallel for
		for(int i = 0;i<(int)currentLevel.size();i++){
			int node = currentLevel[i];

			//only one thread prints at a time
			#pragma omp critical
			cout<<node<< " ";

			//check neighbours
			for(int neighbor : adjList[node]){

				#pragma omp critical
				{
					if(!visited[neighbor]){
						visited[neighbor]= true;
						q.push(neighbor);
					}
				}
			}
		}
	}

	cout<<endl;
}

int main(){

	int edges;

	cout<< "Enter number of vertices:";
	cin>> vertices;

	adjList.resize(vertices);

	cout<< "Enter number of edges:";
	cin>>edges;

	cout<< "Enter the number of edges as u v\n";

	for(int i = 0;i<edges;i++){
		int u,v;
		cin>>u>>v;
		addEdge(u,v);
	}

	int start;

	cout<<"Enter the starting node:";
	cin>>start;

	double t1 = omp_get_wtime();

	parallelBFS(start);

	double t2 = omp_get_wtime();

	cout<<"BFS Time:"<<(t2-t1)<<" sec\n";

	return 0;
}
